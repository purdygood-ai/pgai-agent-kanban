# pgai-agent-kanban: Suggested crontab entries
#
# This is a TEMPLATE file. Before installing, substitute __KANBAN_ROOT__
# with the absolute path to your kanban root directory.
#
# install.sh performs this substitution automatically when you run it.
#
# To install manually:
#   1. Replace all __KANBAN_ROOT__ with your actual kanban root path, e.g.:
#        sed 's|__KANBAN_ROOT__|/home/youruser/pgai_agent_kanban|g' \
#            cron-suggested.template.txt > /tmp/pgai-crontab.txt
#   2. Edit your crontab:
#        crontab -e
#      and paste the contents of /tmp/pgai-crontab.txt (excluding this header)
#
# Cron job logs are written to (kanban-root scope, shared across all projects):
#   __KANBAN_ROOT__/logs/cron-<agent>.log
#
# To disable an agent, comment out its cron line(s) with a leading #.
#
# Schedules summary:
#   PM:                  :03 :08 :13 :55 (every hour)
#   CODER:               :13 :21 :34     (every hour)
#   WRITER:              :13 :21 :34     (every hour)
#   TESTER:              :21 :34 :55     (every hour)
#   CM:                  :34 :55         (every hour)
#   Cleanup (full):      Sundays at 4:00 AM (weekly)
#   Log Cleanup:         Daily at 4:30 AM
#   Trivial-log purge:   Daily at 4:15 AM (cleanup.sh --trivial-only; commented out, opt-in)

# --- Environment ---
PGAI_AGENT_KANBAN_ROOT_PATH=__KANBAN_ROOT__
PATH=__CLAUDE_PATH__:/usr/local/bin:/usr/bin:/bin

# --- PM Agent (Project Manager) ---
# Runs at :03, :08, :13, :55 every hour
3  * * * * __KANBAN_ROOT__/scripts/wake-batch.sh pm  >> __KANBAN_ROOT__/logs/cron-pm.log 2>&1
8  * * * * __KANBAN_ROOT__/scripts/wake-batch.sh pm  >> __KANBAN_ROOT__/logs/cron-pm.log 2>&1
13 * * * * __KANBAN_ROOT__/scripts/wake-batch.sh pm  >> __KANBAN_ROOT__/logs/cron-pm.log 2>&1
55 * * * * __KANBAN_ROOT__/scripts/wake-batch.sh pm  >> __KANBAN_ROOT__/logs/cron-pm.log 2>&1

# --- Coder Agent ---
# Runs at :13, :21, :34 every hour
13 * * * * __KANBAN_ROOT__/scripts/wake-batch.sh coder  >> __KANBAN_ROOT__/logs/cron-coder.log 2>&1
21 * * * * __KANBAN_ROOT__/scripts/wake-batch.sh coder  >> __KANBAN_ROOT__/logs/cron-coder.log 2>&1
34 * * * * __KANBAN_ROOT__/scripts/wake-batch.sh coder  >> __KANBAN_ROOT__/logs/cron-coder.log 2>&1

# --- Writer Agent ---
# Runs at :13, :21, :34 every hour
13 * * * * __KANBAN_ROOT__/scripts/wake-batch.sh writer  >> __KANBAN_ROOT__/logs/cron-writer.log 2>&1
21 * * * * __KANBAN_ROOT__/scripts/wake-batch.sh writer  >> __KANBAN_ROOT__/logs/cron-writer.log 2>&1
34 * * * * __KANBAN_ROOT__/scripts/wake-batch.sh writer  >> __KANBAN_ROOT__/logs/cron-writer.log 2>&1

# --- Tester Agent ---
# Runs at :21, :34, :55 every hour
21 * * * * __KANBAN_ROOT__/scripts/wake-batch.sh tester  >> __KANBAN_ROOT__/logs/cron-tester.log 2>&1
34 * * * * __KANBAN_ROOT__/scripts/wake-batch.sh tester  >> __KANBAN_ROOT__/logs/cron-tester.log 2>&1
55 * * * * __KANBAN_ROOT__/scripts/wake-batch.sh tester  >> __KANBAN_ROOT__/logs/cron-tester.log 2>&1

# --- CM Agent (Configuration Manager / Release) ---
# Runs at :34, :55 every hour
34 * * * * __KANBAN_ROOT__/scripts/wake-batch.sh cm  >> __KANBAN_ROOT__/logs/cron-cm.log 2>&1
55 * * * * __KANBAN_ROOT__/scripts/wake-batch.sh cm  >> __KANBAN_ROOT__/logs/cron-cm.log 2>&1

# --- Cleanup Script (weekly, Sundays at 4:00 AM) ---
0 4 * * 0 __KANBAN_ROOT__/scripts/cleanup/cleanup.sh >> __KANBAN_ROOT__/logs/cron-cleanup.log 2>&1

# --- Trivial-log purge (daily at 04:15, opt-in) ---
# Purges small (< PGAI_CLEANUP_TRIVIAL_LOG_BYTES, default 1700B) AND old
# (> PGAI_CLEANUP_TRIVIAL_LOG_HOURS, default 6h) batch logs from
# __KANBAN_ROOT__/logs/agents/. These accumulate rapidly from cron-driven
# wake-script firings and carry no debugging value after a few hours.
# Per-task logs (tasks/<task_id>/logs/) are NOT affected.
# Uncomment this line to enable daily trivial-log purge:
# 15 4 * * * __KANBAN_ROOT__/scripts/cleanup/cleanup.sh --trivial-only >> __KANBAN_ROOT__/logs/cron-cleanup-trivial.log 2>&1

# --- Log Cleanup (daily at 04:30) ---
# Rotates and prunes old log files to prevent disk bloat
30 4 * * * __KANBAN_ROOT__/scripts/cleanup/cleanup_logs.sh >> __KANBAN_ROOT__/logs/cron-cleanup-logs.log 2>&1
